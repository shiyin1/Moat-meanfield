    \fmfcmd{%
	vardef cross_bar (expr p, len, ang) =
	((-len/2,0)--(len/2,0))
	rotated (ang + angle direction length(p)/2 of p)
	shifted point length(p)/2 of p
	enddef;
	style_def crossed expr p =
	draw (wiggly p);
	ccutdraw cross_bar (p, 10mm, 90);
	enddef;}
    \fmf{crossed,fore=red}{#1,#2}}
